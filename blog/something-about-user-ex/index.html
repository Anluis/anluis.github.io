  if (allScrollLefts.length <= store.groupAndGoods.length) {

      const query = Taro.createSelectorQuery();
      query
        .select(`#swiper-${i}`)
        .boundingClientRect(function (rect) {
        const subLeft = rect.left;
        const subHalfWidth = rect.width / 2;
        const screenHalfwidth = width / 2;
        const needScroll =
              subLeft - screenHalfwidth + subHalfWidth;
        allScrollLefts.push(needScroll);
        if (index === i) {
          setScrollLeft(needScroll);
        }
      })
        .exec();
    }
  } else {

}
